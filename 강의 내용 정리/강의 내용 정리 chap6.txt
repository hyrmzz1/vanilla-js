#6.0 Quotes

js 파일에 array 만들고 각각의 오브젝트로 명언들 넣기. 각각의 object들 콤마로 연결해주기
const quotes = [{
	quote: ,
	author: ,
},
{},{}, ...,{
}]

html 파일에선 quote div 만들고 안에 span 두개를 넣어준다 각각 명언과 저자 들어갈 곳이다!
quote 와 author 는 쿼리셀렉터 이용해 각각의 span 지정하여 정의(const)해줌.

각 array에 접근하는 법? quotes[0] ~ quotes[9]

Math.round()	//float=>int. 반올림.
Math.round(3.4)	//3	Math.round(3.9)	//4
Math.ceil()	//float=>int. 올림.
Math.ceil(1.3)	//2	Math.ceil(1.8)	//2	Math.ceil(1.0)	//1	Math.ceil(1.01)	//2
Math.floor()	//float=>int. 버림.
Math.floor(2.4)	//2	Math.floor(2.899)	//2

randomness(무작위성)
Math.random() * 10	//0부터 10 사이의 숫자를 랜덤으로 얻을 수 있음. 그 숫자는 float
Math.floor(Math.random() * 10) 사용하면 된다.
//그치만 여기에 10이란 숫자 사용하면 array object 개수 늘릴 때마다 수정해줘야함
const todaysQuote = quotes[Math.floor(Math.random() * quotes.length)];

quote 와 author의 innerText에 위의 todaysQuote.quote 와 todaysQuote.author 각각 넣어주면 된다.

#6.1 Background
img 폴더 만들고 배경화면 jpeg들 두기.
js에 images array 만들기. 
const images = ["0.jpeg","1.jpeg","2.jpeg"];	//배열 안 object명과 img 파일명 같게 작성

const chosenImage = images[Math.floor(Math.random() * images.length)];

이 이미지를 실제 html에 추가하려면?!	//js=>html
html만으로는 추가할 수 없고, JS에서 이미지를 만든 후 html에 추가하면 된다.
const bgImage = document.createElement("img");
bgImage.src = `img/${chosenImage}`;	//백틱 이용.	//bgImage.src = "/img"+chosenImage 와 같음.
document.body.appendChild(bgImage);	//appendChild는 body에 html을 가장 뒤에 추가한다.
//prepend()는 appendChild()와 반대로 가장 앞에 추가.

위의 세줄은 html에서의 <img src=""/> 와 같은 작업.
